Zadatak 1
Napisati program kojim se unose sa tastature 3 broja, a zatim pronalazi njihov prosek.
using System; using System.Collections.Generic; using System.Linq; using System.Text; namespace ConsoleApplication1 { class Program { static void Main(string[] args) { float a, b, c, x; string prvi, drugi, treci, prosek; Console.WriteLine("Unesite 3 broja: "); prvi = Console.ReadLine(); drugi = Console.ReadLine(); treci = Console.ReadLine(); a = Convert.ToInt32(prvi); b = Convert.ToInt32(drugi); c = Convert.ToInt32(treci); x = (a + b + c) / 3; prosek = Convert.ToString(x); Console.WriteLine("Prosek iznosi:" +prosek); Console.ReadKey(); } } }

Zadatak 3
Sastaviti konzolnu aplikaciju, gde se unose dva broja i štampa njihov zbir i razlika. Napraviti klasu matematika, i u okviru nje dve funkcije: za sabiranje i oduzimanje.
using System; using System.Collections.Generic; using System.Linq; using System.Text; namespace ConsoleApplication3 { class Program { static void Main(string[] args) { Matematika M = new Matematika(); Console.WriteLine("Uneti dva broja: "); string prvi=Console.ReadLine(); string drugi=Console.ReadLine(); double x=Convert.ToDouble(prvi); double y=Convert.ToDouble(drugi); Console.WriteLine ( "Zbir je"+M.Zbir (x, y) ); Console.WriteLine ("Razlika je"+ M.Razlika (x, y) ); Console.ReadKey(); } } class Matematika //definisali smo klasu matematika { public double Zbir(double A, double B) { return A + B; } public double Razlika(double A, double B) { return A - B; } } }
